# Block UI

```html
<sc-block-ui></sc-block-ui>
```

This component lets you block the ui of an interface while loading.

import DemoTabs from '../../components/Demo-Tabs.js';
import ComponentDoc from '../../components/Component-Doc.js';

<DemoTabs>
	<DemoTabs.Code>
{`
<div style="position: relative">
	<sc-address></sc-address>
	<sc-block-ui spinner></sc-block-ui>
</div>
`}
	</DemoTabs.Code>
	<div style={{position :'relative'}}>
		<sc-address></sc-address>
		<sc-block-ui spinner></sc-block-ui>
	</div>
</DemoTabs>
