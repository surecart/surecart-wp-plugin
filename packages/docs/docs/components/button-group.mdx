# Button Group

```html
<sc-button-group></sc-button-group>
```

This component let users create radio group for holding multiple radio option.

import DemoTabs from '../../components/Demo-Tabs.js';
import ComponentDoc from '../../components/Component-Doc.js';

<DemoTabs>
	<DemoTabs.Code>
{`
<sc-button-group label="Choose an Option">
	<sc-button type="primary">Primary</sc-button>
	<sc-button type="success">Success</sc-button>
	<sc-button type="info">Info</sc-button>
</sc-button-group>
`}
	</DemoTabs.Code>
	<sc-button-group label="Choose an Option">
		<sc-button type="primary">Primary</sc-button>
		<sc-button type="success">Success</sc-button>
		<sc-button type="info">Info</sc-button>
	</sc-button-group>
</DemoTabs>
