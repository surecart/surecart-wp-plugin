# Breadcrumb

```html
<sc-breadcrumb></sc-breadcrumb>
```

This component let users create breadcrumb.

:::info
Breadcrumb components must be wrapped in the `<sc-breadcrumbs></sc-breadcrumbs>` component.
:::

import DemoTabs from '../../components/Demo-Tabs.js';
import ComponentDoc from '../../components/Component-Doc.js';

<DemoTabs>
	<DemoTabs.Code>
{`
<sc-breadcrumbs>
	<sc-breadcrumb>First</sc-breadcrumb>
	<sc-breadcrumb>Second</sc-breadcrumb>
</sc-breadcrumbs>
`}
	</DemoTabs.Code>
	<sc-breadcrumbs>
		<sc-breadcrumb>First</sc-breadcrumb>
		<sc-breadcrumb>Second</sc-breadcrumb>
	</sc-breadcrumbs>
</DemoTabs>
