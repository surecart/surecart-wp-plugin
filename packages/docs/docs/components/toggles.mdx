# Toggles

```html
<sc-toggles></sc-toggles>
```

This component let users create toggles container.

import DemoTabs from '../../components/Demo-Tabs.js';
import ComponentDoc from '../../components/Component-Doc.js';

<DemoTabs>
	<DemoTabs.Code>
{`
<sc-toggles summary="Toggle Summary" showControl="true">
	<sc-toggle summary="Toggle Summary" showControl="true">My Toggle Text</sc-toggle>
	<sc-toggle summary="Toggle Summary" showControl="true">My Toggle Text</sc-toggle>
	<sc-toggle summary="Toggle Summary" showControl="true">My Toggle Text</sc-toggle>
</sc-toggles>
`}
	</DemoTabs.Code>
	<sc-toggles summary="Toggle Summary" showControl="true">
		<sc-toggle summary="Toggle Summary" showControl="true">My Toggle Text</sc-toggle>
		<sc-toggle summary="Toggle Summary" showControl="true">My Toggle Text</sc-toggle>
		<sc-toggle summary="Toggle Summary" showControl="true">My Toggle Text</sc-toggle>
	</sc-toggles>
</DemoTabs>
