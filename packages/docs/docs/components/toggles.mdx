# Toggles

```html
<sc-toggles></sc-toggles>
```

This component let users create toggles container.

import DemoTabs from '../../components/Demo-Tabs.js';
import ComponentDoc from '../../components/Component-Doc.js';

<DemoTabs>
	<DemoTabs.Code>
{`
<sc-toggles theme="container" accordion>
	<sc-toggle summary="First Option" borderless shady>My Toggle Text</sc-toggle>
	<sc-toggle summary="Second Option" borderless shady>My Toggle Text</sc-toggle>
	<sc-toggle summary="Third Option" borderless shady>My Toggle Text</sc-toggle>
</sc-toggles>
`}
	</DemoTabs.Code>
	<sc-toggles theme="container" accordion>
		<sc-toggle summary="First Option" borderless shady>My Toggle Text</sc-toggle>
		<sc-toggle summary="Second Option" borderless shady>My Toggle Text</sc-toggle>
		<sc-toggle summary="Third Option" borderless shady>My Toggle Text</sc-toggle>
	</sc-toggles>
</DemoTabs>
