# Select

```html
<sc-select></sc-select>
```

This component let users create select container.

import DemoTabs from '../../components/Demo-Tabs.js';
import ComponentDoc from '../../components/Component-Doc.js';
import {ScSelect} from '@surecart/components-react';

<DemoTabs>
	<DemoTabs.Code>
{`
<sc-select label="Select" placeholder="Select something" search></sc-select>
<script>
	document.querySelector('sc-select').choices = [
		{label: 'Item', value: 'item-1'},
		{label: 'Item 2', value: 'item-2'}
	];
</script>
`}
	</DemoTabs.Code>
	<ScSelect label="Select" placeholder="Select something" search choices={[{label: 'Item', value: 'item-1'}, {label: 'Item 2', value: 'item-2'}]} />
</DemoTabs>
