# Select

```html
<ce-select></ce-select>
```

This component lets the user select one option from a group of many.

import DemoTabs from '../../components/Demo-Tabs.js';
import ComponentDoc from '../../components/Component-Doc.js';
import { CeSelect } from '@surecart/components-react';

<DemoTabs>
  <DemoTabs.Code>
    {`
<ce-select id="my-select"></ce-select>
<script>
  var select = document.querySelector('#my-select');
  // add choices
  select.choices = [
    { "value": "pizza", "label": "Pizza", "selected": false, "disabled": false },
    { "value": "pasta", "label": "Pasta", "selected": false, "disabled": false },
    { "value": "breadsticks", "label": "Breadsticks", "selected": false, "disabled": false },
    { "value": "no-stock", "label": "Someting out of stock.", "selected": false, "disabled": true },
  ];
</script>
    `}

  </DemoTabs.Code>

  <div></div>
	<CeSelect placeholder="Make your choice" placeholderValue="Make your choice" choices={[
    { "value": "pizza", "label": "Pizza", "selected": false, "disabled": false },
    { "value": "pasta", "label": "Pasta", "selected": false, "disabled": false },
    { "value": "breadsticks", "label": "Breadsticks", "selected": false, "disabled": false },
    { "value": "no-stock", "label": "Someting out of stock.", "selected": false, "disabled": true },
  ]}></CeSelect>
</DemoTabs>
