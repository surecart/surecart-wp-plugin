# Switch

```html
<presto-switch></presto-switch>
```

import DemoTabs from "../../components/Demo-Tabs.js";

<DemoTabs>
  <DemoTabs.Code>{"<presto-switch>Toggle Me</presto-switch>"}</DemoTabs.Code>
  <presto-switch>Toggle Me</presto-switch>
</DemoTabs>

## Disabled

Use the `disabled` property to disable the switch

<DemoTabs>
  <DemoTabs.Code>
    {"<presto-switch disabled>Disabled</presto-switch>"}
  </DemoTabs.Code>
  <presto-switch disabled>Disabled</presto-switch>
</DemoTabs>

## Custom Size

Use the `disabled` property to disable the switch

<DemoTabs>
  <DemoTabs.Code>
    {
      '<presto-switch style="--width: 80px; --height: 32px; --thumb-size: 26px;"></presto-switch>'
    }
  </DemoTabs.Code>
  <presto-switch
    style={{ "--width": "80px", "--height": "32px", "--thumb-size": "26px" }}
  ></presto-switch>
</DemoTabs>
